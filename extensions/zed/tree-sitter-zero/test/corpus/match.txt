fun boolScore(value: Bool) -> i32 {
    match value {
        true { return 1 }
        false { return 2 }
        ._ { return 0 }
    }
    return 0
}

fun bucket(value: u8, enabled: Bool) -> i32 {
    match value {
        0 if enabled { return 10 }
        1..3 { return 20 }
    }
    return 0
}
